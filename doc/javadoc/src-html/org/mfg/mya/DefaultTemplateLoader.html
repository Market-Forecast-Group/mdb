001package org.mfg.mya;
002
003import java.io.InputStream;
004
005/**
006 * Default implementation of {@link ITemplateLoader}. The templates are loaded
007 * from the resources of a given class, with the
008 * {@link Class#getResourceAsStream(String)}.
009 * 
010 * @author arian
011 *
012 */
013public class DefaultTemplateLoader implements ITemplateLoader {
014        private Class<?> _resourcesClass;
015
016        /**
017         * Create a default template loader.
018         * 
019         * @param resourcesClass
020         *            The class used to load the resources.
021         * @see Class#getResourceAsStream(String)
022         */
023        public DefaultTemplateLoader(Class<?> resourcesClass) {
024                _resourcesClass = resourcesClass;
025        }
026
027        /**
028         * The class used to load the resources as streams.
029         * 
030         * @return The class used to load the resources.
031         */
032        public Class<?> getResourcesClass() {
033                return _resourcesClass;
034        }
035
036        /**
037         * Load the template from the resources of the {@link #getResourcesClass()}.
038         * Is the same of:
039         * 
040         * <pre>
041         * getResourcesClass().getResourceAsStream(name);
042         * </pre>
043         * 
044         * @param name
045         *            The name of the resource/template.
046         */
047        @Override
048        public InputStream load(String name) {
049                return _resourcesClass.getResourceAsStream(name);
050        }
051
052}
