001package org.mfg.mya;
002
003import java.io.InputStream;
004
005/**
006 * <p>
007 * Interface to abstract the template loading. A template is loaded when the
008 * with instruction:
009 * </p>
010 * 
011 * <pre>
012 * #mya:import &lt;template-name&gt;
013 * </pre>
014 * <p>
015 * A loader implementation build an input stream for the given template name.
016 * </p>
017 * 
018 * @author arian
019 *
020 */
021public interface ITemplateLoader {
022
023        /**
024         * Build an input stream for the given template name.
025         * 
026         * @param name
027         *            The name of the template.
028         * @return An input stream with the template identified with that name.
029         */
030        public InputStream load(String name);
031}
