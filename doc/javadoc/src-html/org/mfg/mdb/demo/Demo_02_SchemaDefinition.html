001package org.mfg.mdb.demo;
002
003import java.io.IOException;
004
005import org.mfg.mdb.compiler.Compiler;
006import org.mfg.mdb.compiler.Order;
007import org.mfg.mdb.compiler.Type;
008
009/**
010 * <p>
011 * Define a schema to store prices. It contains one "Price" table with "time"
012 * and "price" columns.
013 * </p>
014 * <p>
015 * You can follow this class as a template for you to define your own schemas.
016 * Note the different elements:
017 * </p>
018 * <ol>
019 * <li>Your class should extends the {@link Compiler} class.</li>
020 * <li>In the constructor you code the schema using the "table" and "column"
021 * methods.</li>
022 * <li>In the main method you execute the compiler.</li>
023 * </ol>
024 * <p>
025 * <a href=
026 * "../../../../src-html/org/mfg/mdb/demo/Demo_02_SchemaDefinition.html#line.33"
027 * > See the code</a>
028 * </p>
029 * 
030 * @author arian
031 *
032 */
033public class Demo_02_SchemaDefinition extends Compiler {
034
035        /**
036         * In the constructor you define the schema.
037         */
038        {
039                schemaName("Demo_02_");
040                packageName("org.mfg.mdb.demo");
041
042                // start a table definition
043                table("Demo_02_Price");
044                // define a "time" column of type "int" with an ascending order
045                // the order will be used to generate special methods like binary search
046                // methods.
047                column("time", Type.INTEGER, Order.ASCENDING);
048
049                // define a "price" column of type "long".
050                column("price", Type.LONG);
051        }
052
053        /**
054         * In the main method you execute the compiler.
055         * 
056         * @param args
057         *            The program arguments.
058         * @throws IOException
059         *             If there is any I/O error.
060         */
061        public static void main(String[] args) throws IOException {
062                Demo_02_SchemaDefinition demo = new Demo_02_SchemaDefinition();
063                String pathToSrcFolder = "/home/arian/Documents/Source/mdb/projects/org.mfg.mdb.demo/src";
064                demo.compile(pathToSrcFolder);
065        }
066}
