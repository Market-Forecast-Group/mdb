011package org.mfg.mdb.runtime;
012
013/**
014 * Used by validators to report errors.
015 * 
016 * @see IValidator
017 * @author arian
018 * @param <T>
019 *            The generated record class.
020 * 
021 */
022public interface IValidatorListener<T extends IRecord> {
023
024        /**
025         * This method is called by a validator if there is any error.
026         * 
027         * @param args
028         *            The error information.
029         */
030        public void errorReported(ValidatorError<T> args);
031}
