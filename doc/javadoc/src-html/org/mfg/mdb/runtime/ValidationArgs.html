011package org.mfg.mdb.runtime;
012
013/**
014 * Arguments of a validation method.
015 * 
016 * @author arian
017 * @param <T>
018 *            The generated record class.
019 * 
020 */
021public class ValidationArgs<T extends IRecord> {
022        private MDB<T> _mdb;
023        private T _prev;
024        private T _current;
025        private long _row;
026
027        /**
028         * The constructor.
029         * 
030         * @param mdb
031         *            The MDB instance in question.
032         * @param row
033         *            The index of the current record.
034         * @param prev
035         *            The previous record.
036         * @param current
037         *            The current record.
038         */
039        public ValidationArgs(MDB<T> mdb, long row, T prev, T current) {
040                super();
041                _mdb = mdb;
042                _prev = prev;
043                _current = current;
044                _row = row;
045        }
046
047        /**
048         * MDB instance is validating.
049         * 
050         * @return The MDB instance.
051         */
052        public MDB<T> getMdb() {
053                return _mdb;
054        }
055
056        /**
057         * Set the MDB instance is validating.
058         * 
059         * @param mdb
060         *            The MDB instance.
061         */
062        public void setMdb(MDB<T> mdb) {
063                this._mdb = mdb;
064        }
065
066        /**
067         * Get the previous record.
068         * 
069         * @return The generated record class.
070         * 
071         */
072        public T getPrev() {
073                return _prev;
074        }
075
076        /**
077         * Set the previous record.
078         * 
079         * @param prev
080         *            The previous record.
081         */
082        public void setPrev(T prev) {
083                this._prev = prev;
084        }
085
086        /**
087         * Get the current record.
088         * 
089         * @return The generated record class.
090         */
091        public T getCurrent() {
092                return _current;
093        }
094
095        /**
096         * Set the current record.
097         * 
098         * @param current
099         *            The current record.
100         */
101        public void setCurrent(T current) {
102                this._current = current;
103        }
104
105        /**
106         * Get the current record position.
107         * 
108         * @return The index of the current record (or row).
109         */
110        public long getRow() {
111                return _row;
112        }
113
114        /**
115         * Set the current record position.
116         * 
117         * @param row
118         *            The index of the current record (or row).
119         */
120        public void setRow(long row) {
121                this._row = row;
122        }
123}
