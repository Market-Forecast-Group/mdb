001package org.mfg.mdb.runtime;
002
003import java.io.IOException;
004
005/**
006 * Checked exception thrown when an attempt to move a closed cursor.
007 * 
008 * @author arian
009 * 
010 */
011public class ClosedCursorException extends IOException {
012        private static final long serialVersionUID = 1L;
013
014        /**
015         * The constructor
016         * 
017         * @param cursor
018         *            The cursor.
019         */
020        public ClosedCursorException(ICursor<?> cursor) {
021                super("Attempt to move a closed cursor of " + cursor.getMDB().getFile());
022        }
023}
