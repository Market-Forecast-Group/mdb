011package org.mfg.mdb.runtime;
012
013import java.io.IOException;
014
015/**
016 * Common interface for all the appenders.
017 * 
018 * @author arian
019 * @param <T>
020 *            The generated record type.
021 * 
022 */
023public interface IAppender<T extends IRecord> {
024        /**
025         * Close the associated resources.
026         * 
027         * @throws IOException
028         *             If there is any problem closing the files.
029         */
030        public void close() throws IOException;
031
032        /**
033         * Get the associated MDB instance.
034         * 
035         * @return The MDB instance.
036         */
037        public MDB<T> getMDB();
038
039        /**
040         * Add a new record (formed by the appender values) to the file.
041         * 
042         * @throws IOException
043         *             If there is any problem writing the files.
044         */
045        public void append() throws IOException;
046
047        /**
048         * Add to the file the given <code>record</code>.
049         * 
050         * @param record
051         *            Record to add.
052         * @throws IOException
053         *             If there is any problem writing the files.
054         */
055        public void append(T record) throws IOException;
056
057        /**
058         * Add to the file the given record. Warning, the same record instance is
059         * inserted in the memory buffer, do not modify it after this operation.
060         * Just use this method if you know what are you doing.
061         * 
062         * @param record
063         *            The record to append.
064         * @throws IOException
065         *             If there is any problem writing the file.
066         */
067        public void append_ref_unsafe(T record) throws IOException;
068
069        /**
070         * Create a record with the appender's values.
071         * 
072         * @return The record.
073         */
074        public T toRecord();
075}
