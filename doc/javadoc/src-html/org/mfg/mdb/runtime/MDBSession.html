087        /**
088         * Create a new session.
089         * 
090         * @param sessionName
091         *            Session name, used only as a name.
092         * @param root
093         *            The database folder.
094         * @param mode
095         *            Session mode used to assert a correct use of the session.
096         * @param signatures
097         *            The signature of the schema, used to validate if the database
098         *            is readable by this session. This information is generated by
099         *            the compiler.
100         * @param jsonSchema
101         *            The schema in JSON format. It is generated by the compiler and
102         *            saved into the database, then external tools can read the
103         *            files without create a session.
104         * @throws IOException
105         *             If there is any problem writing/reading the session files. If
106         *             there is any problem writing/reading session files.
107         */
108        public MDBSession(final String sessionName, final File root,
109                        final SessionMode mode, Map<String, String> signatures,
110                        String jsonSchema) throws IOException {
111                _debug = false;
112                _memory = mode == SessionMode.MEMORY;
113                _signatures = signatures;
114                _jsonSchema = jsonSchema;
115
116                if (mode == SessionMode.READ_ONLY) {
117                        if (!root.exists()) {
118                                throw new IOException(
119                                                "Session mode READ_ONY requires a previous database exists");
120                        }
121                } else if (mode == SessionMode.READ_WRITE) {
122                        root.mkdirs();
123                        if (!root.exists()) {
124                                throw new IOException("Fail create root dir at " + root);
125                        }
126                }
127
128                _sessionName = sessionName;
129                _root = root;
130                _backupDir = _root.toPath().resolveSibling("~" + root.getName())
131                                .toFile();
132                _mode = mode;
133                _cache = new HashMap<>();
134                _open = true;
135                _properties = new Properties();
136
137                _deferredCursors = new ArrayList<>();
138
139                if (!_memory) {
140                        initProperties();
141                        initMetadata();
142                }
143                ReentrantReadWriteLock lock = new ReentrantReadWriteLock();
144                _readLock = lock.readLock();
145                _writeLock = lock.writeLock();
146                _allCursorsClosedCond = _writeLock.newCondition();
147
148                _closeTimeoutMillis = 3_000;
149
150                _openCursors = 0;
151        }

159                }

162        private void initMetadata() throws IOException {
163                _metadataDir = new File(_root, ".metadata");
164                if (!_metadataDir.exists()) {
165                        if (!_metadataDir.mkdirs()) {
166                                throw new IOException("Cannot create metadata dir at "
167                                                + _metadataDir);
168                        }
169                }
170
171                checkCompatibility(_signatures);
172                Path schemaFile = getMetadataDir().toPath().resolve("schema.json");
173                Files.write(schemaFile, _jsonSchema.getBytes());
174
175                _filesTable = new FilesTable(_root.toPath());
176        }

205        /**
206         * Check if the current MDB generated classes are compatible with the
207         * database. Maybe it is an old database but the current schema is other.
208         * 
209         * @param signatures
210         * @throws IOException
211         *             If there is any problem writing/reading the session files.
212         */
213        protected void checkCompatibility(Map<String, String> signatures)
214                        throws IOException {
215                Properties props = new Properties();
216                File file = new File(_metadataDir, "signatures.properties");
217                if (file.exists()) {
218                        try (FileInputStream in = new FileInputStream(file)) {
219                                props.load(in);
220                                in.close();
221                                for (Object tableId : props.keySet()) {
222                                        if (signatures.containsKey(tableId)) {
223                                                String newSignature = signatures.get(tableId);
224                                                Object oldSignature = props.get(tableId);
225                                                if (!newSignature.equals(oldSignature)) {
226                                                        throw new IOException("Wrong signature for table "
227                                                                        + tableId);
228                                                }
229                                        }
230                                }
231                        }
232                }
233                props.clear();
234                for (String key : signatures.keySet()) {
235                        props.put(key, signatures.get(key));
236                }
237                try (FileOutputStream output = new FileOutputStream(file)) {
238                        props.store(output, null);
239                }
240        }

285         * This method is called by MDB generated classes. Do not call it yourself.
286         * 
287         * @param cursor
288         *            The cursor.
289         * 
290         */
291        public void cursorCreated(final ICursor<?> cursor) {
292                if (_debug) {
293                        out.println("Cursor created " + cursor + " - "
294                                        + cursor.getMDB().getFile());
295                }
296                _writeLock.lock();
297                _openCursors++;
298                try {
299                        int size = _deferredCursors.size();
300                        if (size > DEFERRED_CURSORS_THRESHOLD) {
301
302                                List<ICursor<?>> list = new ArrayList<>();
303                                for (ICursor<?> c : _deferredCursors) {
304                                        if (c != null && !c.isOpen()) {
305                                                list.add(c);
306                                        }
307                                }
308                                _deferredCursors.removeAll(list);
309                                if (_debug) {
310                                        out.println("Reduce list of deferred cursors from " + size
311                                                        + " to " + _deferredCursors.size());
312                                }
313                        }
314                } finally {
315                        _writeLock.unlock();
316                }
317        }

326        public void cursorClosed(ICursor<?> cursor) {
327                if (_debug) {
328                        out.println("Cursor closed " + cursor + " - "
329                                        + cursor.getMDB().getFile());
330                }
331                _writeLock.lock();
332                _openCursors--;
333                assert _openCursors >= 0;
334                try {
335                        _allCursorsClosedCond.signalAll();
336                } finally {
337                        _writeLock.unlock();
338                }
339        }

341        /**
342         * Defer a cursor. Deferred cursors are closed automatically before to close
343         * the session. This is useful for cursors you pretend to use in the whole
344         * session live. However please take a look to the methods
345         * {@link MDB#thread_cursor()} and {@link MDB#thread_randomCursor()},
346         * probably a reusable thread cursor is what you need.
347         * 
348         * @see MDB#thread_cursor()
349         * @see MDB#thread_randomCursor()
350         * 
351         * @param cursor
352         *            The cursor to close automatically at the end of the session.
353         */
354        public void defer(final ICursor<?> cursor) {
355                if (cursor == null) {
356                        throw new IllegalArgumentException("Cannot defer a null cursor.");
357                }
358
359                _writeLock.lock();
360                try {
361                        _deferredCursors.add(cursor);
362                } finally {
363                        _writeLock.unlock();
364                }
365        }

367        /**
368         * <p>
369         * Disconnect the MDB file from the session. When you disconnect an MDB
370         * file, the appender and cursors are closed, so all the resources
371         * associated to that file are disposed.
372         * </p>
373         * <p>
374         * This method is useful for those databases with a big number of files,
375         * maybe at certain moment the application needs to read some information of
376         * from the files but later only a small portion of them is accessed, in
377         * this case the application can disconnect the not needed files.
378         * </p>
379         * 
380         * @param mdb
381         *            MDB file to disconnect.
382         * @throws IOException
383         *             If there is any problem writing/reading the session files.
384         * @throws TimeoutException
385         *             This exception is sent if the session is busy. Check there is
386         *             not any open cursor before to call this method.
387         */
388        public void disconnect(MDB<?> mdb) throws IOException, TimeoutException {
389                _writeLock.lock();
390                try {
391                        mdb.closeAppender();
392                        closeCursors(mdb);
393                        _cache.remove(mdb);
394                } finally {
395                        _writeLock.unlock();
396                }
397        }

399        /**
400         * Close the cursors.
401         * 
402         * @param mdb
403         *            The owner of the cursors, or <code>null</code> to close all
404         *            the cursors of the session.
405         * @throws TimeoutException
406         *             This exception is sent if the session is busy. Check there is
407         *             not any open cursor before to call this method.
408         * @throws IOException
409         *             If there is any problem writing/reading the session files.
410         */
411        void closeCursors(MDB<?> mdb) throws TimeoutException, IOException {
412                // close the deferred cursors
413                for (ICursor<?> cur : _deferredCursors) {
414                        if (cur != null) {
415                                if (mdb == null || cur.getMDB() == mdb) {
416                                        // we take the risk to close the cursor
417                                        // and we hope the user will not it use
418                                        // again.
419                                        cur.close();
420                                }
421                        }
422                }
423                if (_closeTimeoutMillis > 0) {
424                        long t = currentTimeMillis();
425                        if (_debug) {
426                                out.println("Waiting for cursors to close the session...");
427                        }
428
429                        // wait for the reading threads to close the cursors
430                        // this "waiting" help to applications
431                        // where is difficult to synchronize the reading threads
432                        // with the one who closes the session.
433                        int count;
434                        while (true) {
435                                if (mdb == null) {
436                                        count = getOpenCursorCount();
437                                } else {
438                                        count = mdb.getOpenCursorCount();
439                                }
440                                if (count == 0) {
441                                        break;
442                                }
443
444                                assert count >= 0 : "Bug: negative number of open cursors ("
445                                                + count + ")";
446
447                                try {
448                                        if (_debug) {
449                                                out.println(this + " await...");
450                                        }
451                                        _allCursorsClosedCond.await(10, TimeUnit.MILLISECONDS);
452                                        if (currentTimeMillis() - t > _closeTimeoutMillis) {
453                                                if (_debug) {
454                                                        for (MDB<?> mdb2 : _cache.values()) {
455                                                                int count2 = mdb2.getOpenCursorCount();
456                                                                if (count2 != 0) {
457                                                                        out.println(mdb2 + " open cursors: "
458                                                                                        + count2 + " - " + mdb2.getFile());
459                                                                }
460                                                        }
461                                                }
462                                                throw new TimeoutException("Waiting for " + count
463                                                                + " cursors to close the session.");
464                                        }
465                                } catch (InterruptedException e) {
466                                        e.printStackTrace();
467                                }
468                        }
469                        if (_debug) {
470                                out.println(this + " All cursors are closed, waiting time: "
471                                                + (currentTimeMillis() - t) + " - " + getRoot());
472                        }
473                }
474
475        }

651        /**
652         * Visitor to copy a tree of files.
653         * 
654         * @author arian
655         * 
656         */
657        static class CopyVisitor extends SimpleFileVisitor<Path> {
658                private Path _from;
659                private Path _to;
660
661                public CopyVisitor(Path from, Path to) {
662                        super();
663                        _from = from;
664                        _to = to;
665                }
666
667                @Override
668                public FileVisitResult visitFile(Path file, BasicFileAttributes attrs)
669                                throws IOException {
670                        Path rel = _from.relativize(file);
671                        Path target = _to.resolve(rel);
672                        Files.createDirectories(target.getParent());
673                        Files.copy(file, target, StandardCopyOption.REPLACE_EXISTING);
674                        return FileVisitResult.CONTINUE;
675                }
676        }

693                try {
694                        flush();
695                        saveProperties();
696
697                        BackupVersion version = backup(_root, comment);

732        public BackupVersion backupFile(File mdbFile, String comment)

734                if (_memory)
735                        throw new UnsupportedOperationException(
736                                        "Backup of memory databases are not implemented yet.");
737
738                _writeLock.lock();
739                try {
740                        if (!mdbFile.toPath().toAbsolutePath()
741                                        .startsWith(_root.toPath().toAbsolutePath())) {
742                                throw new IOException("The file " + mdbFile
743                                                + " does not belong to this database: " + _root);
744                        }
745                        MDB<?> mdb = _cache.get(mdbFile);
746                        if (mdb != null) {
747                                mdb.flushAppender();
748                        }
749                        BackupVersion version = backupFile(_root, mdbFile, comment);
750                        return version;
751                } finally {
752                        _writeLock.unlock();
753                }
754        }

756        /**
757         * Backup an MDB file. Use this method if there is not an active session. In
758         * other case, use the session method {@link #backupFile(File, String)} or
759         * {@link #backupFile(MDB, String)}.
760         * 
761         * @param dbRoot
762         *            The root of the database.
763         * @param mdbFile
764         *            The file to backup.
765         * @param comment
766         *            The description of the backup.
767         * @return The backup reference.
768         * @throws IOException
769         *             If there is any problem writing/reading the session files.
770         */
771        public static BackupVersion backupFile(File dbRoot, File mdbFile,
772                        String comment) throws IOException {
773
774                // mdRoot
775                File arrayFile = new File(mdbFile.getPath() + ".array");
776                Path relPath = dbRoot.toPath().relativize(mdbFile.toPath());
777
778                Properties props = new Properties();
779                props.put(KEY_VERSION_PARTIAL_BACKUP, relPath.toString());
780
781                BackupVersion version = createVersion(dbRoot, comment, props, true);
782
783                Path versionedFile = version.getPath().resolve(relPath);
784
785                Files.copy(mdbFile.toPath(), versionedFile);
786
787                if (arrayFile.exists()) {

794                return version;

848        /**
849         * <p>
850         * Restore the session to the given version. Be careful because the data of
851         * the database will be deleted and you cannot recover it. If you want, you
852         * can create a backup before restore a backup.
853         * </p>
854         * <p>
855         * Before to perform the copy, all the deferred cursors will be closed, but
856         * if there are a common cursor open, then an {@link IOException} exception
857         * is sent, so, be sure you closed all the cursors before. Take in
858         * consideration that when you restore a backup, maybe there are files that
859         * are not present in the current session, so the associated MDB will be
860         * invalid and removed from the session cache.
861         * </p>
862         * This method fails (it sends a {@link FileNotFoundException}) if the
863         * backup version was created only for one file (
864         * {@link #backupFile(File, String)} ) and the original file does not exist
865         * any more.
866         * 
867         * @param version
868         *            Backup version to restore.
869         * @throws IOException
870         *             If there is any problem writing/reading the session files.
871         * @throws TimeoutException
872         *             This exception is sent if the session is busy. Check there is
873         *             not any open cursor before to call this method.
874         */
875        public void restore(BackupVersion version) throws IOException,
876                        TimeoutException {
877                _writeLock.lock();
878                try {
879                        if (version.isPartial()) {
880                                restorePartial(_root, _cache, _deferredCursors, version);
881                        } else {
882                                for (MDB<?> mdb : _cache.values()) {
883                                        mdb.disconnectFile();
884                                }
885
886                                closeCursors(null);
887
888                                for (MDB<?> mdb : _cache.values()) {
889                                        if (mdb.getOpenCursorCount() > 0) {
890                                                throw new IOException(
891                                                                "Cannot restore the backup, the file "
892                                                                                + mdb.getFile()
893                                                                                + " is locked by opened cursors.");
894                                        }
895                                }
896
897                                restoreDatabase(_root, version);
898
899                                for (MDB<?> mdb : new ArrayList<>(_cache.values())) {
900                                        if (!mdb.getFile().exists()) {
901                                                err.println("The file "
902                                                                + mdb.getFile()
903                                                                + " is not present in the backup version "
904                                                                + version.getDate()
905                                                                + " - "
906                                                                + version.getComment()
907                                                                + ", from that reason it was removed from the session cache and if the user performs any operation on that file a FileNotFoundException will be sent.");
908                                                _cache.remove(mdb.getFile());
909                                        } else {
910                                                mdb.reconnectFile();
911                                        }
912                                }
913
914                                _properties.clear();
915                                try (InputStream in = Files.newInputStream(_propsFile.toPath())) {
916                                        _properties.load(in);
917                                }
918                                readProperties();
919                        }
920                } finally {
921                        _writeLock.unlock();
922                }
923        }

944        private static void restorePartial(File dbRoot, Map<File, MDB<?>> mdbCache,
945                        List<ICursor<?>> deferredCursors, BackupVersion version)
946                        throws IOException {
947                Properties props = readBackupProperties(version);
948
949                String filename = props.getProperty(KEY_VERSION_PARTIAL_BACKUP);
950
951                Path dbFile = dbRoot.toPath().resolve(filename);
952
953                MDB<?> mdb = mdbCache == null ? null : mdbCache.get(dbFile.toFile());
954                if (mdb != null) {
955                        for (ICursor<?> cursor : deferredCursors) {
956                                if (cursor != null) {
957                                        if (cursor.getMDB() == mdb) {
958                                                cursor.close();
959                                        }
960                                }
961                        }
962
963                        if (mdb.getOpenCursorCount() > 0) {
964                                throw new IOException("Cannot restore the backup, the file "
965                                                + mdb.getFile() + " is locked by opened cursors.");
966                        }
967
968                        mdb.disconnectFile();
969                }
970
971                if (!Files.exists(dbFile)) {
972                        throw new FileNotFoundException(
973                                        "File restore failed, the original file (" + dbFile
974                                                        + ") does not exist.");
975                }
976
977                Path backupFile = version.getPath().resolve(filename);
978                Files.copy(backupFile, dbFile, StandardCopyOption.REPLACE_EXISTING);
979
980                String arrayFileName = backupFile.getFileName() + ".array";
981                Path backupArrayfile = backupFile.resolveSibling(arrayFileName);
982                if (Files.exists(backupArrayfile)) {
983                        Path dbArrayFile = dbFile.resolve(arrayFileName);
984
985                        // not always the array file exists, if the user does not add any
986                        // array element, the array file is not created.
987                        if (Files.exists(dbArrayFile)) {
988                                Files.copy(backupArrayfile, dbArrayFile,
989                                                StandardCopyOption.REPLACE_EXISTING);
990                        }
991                }
992
993                if (mdb != null) {
994                        mdb.reconnectFile();
995                }
996        }

1079                if (_debug) {
1080                        out.println("Closing " + this + " - " + getRoot());

1085                                _open = false;
1086
1087                                if (_filesTable != null) {
1088                                        _filesTable.close();

1092

1102                        }
1103                } catch (Exception e) {
1104                        _open = true;
1105                        throw e;
1106                } finally {
1107                        _writeLock.unlock();
1108                }

1143        /**
1144         * Close the session and delete all the files. If there are any cursor open,
1145         * is possible this method will fail, specially in Windows operating
1146         * systems.
1147         * 
1148         * @return The number of files cannot be deleted.
1149         * @throws IOException
1150         *             If there is any problem writing/reading the session files. If
1151         *             there is any problem closing/deleting the session files.
1152         * @throws TimeoutException
1153         *             Exception sent if the session is busy and cannot be closed.
1154         *             If you get this exception it means there are open cursors you
1155         *             need to close before to call this method.
1156         */
1157        public int closeAndDelete() throws IOException, TimeoutException {
1158                _writeLock.lock();
1159                try {
1160                        if (_cache == null) {
1161                                return 0;
1162                        }
1163
1164                        close(false);
1165
1166                        int fail = 0;
1167                        final Collection<MDB<?>> mdbList = _cache.values();
1168
1169                        for (final MDB<?> mdb : mdbList) {
1170                                if (!mdb.deleteFiles()) {
1171                                        fail++;
1172                                }
1173                        }
1174                        _cache = null;
1175
1176                        fail += delete(new File(_root, "session.properties"));
1177
1178                        if (_root != null) {
1179                                fail += delete(_root);
1180                        }
1181
1182                        if (_backupDir.exists()) {
1183                                fail += delete(_backupDir);
1184                        }
1185
1186                        return fail;
1187                } finally {
1188                        _writeLock.unlock();
1189                }
1190        }
