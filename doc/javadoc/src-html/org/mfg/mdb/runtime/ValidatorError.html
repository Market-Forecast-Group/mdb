011package org.mfg.mdb.runtime;
012
013/**
014 * Contains the information of a validation error.
015 * 
016 * @author arian
017 * @param <T>
018 *            The generated record class.
019 * 
020 */
021public class ValidatorError<T extends IRecord> {
022        private ValidationArgs<T> _args;
023        private String _message;
024
025        /**
026         * The constructor.
027         * 
028         * @param args
029         *            The validation arguments.
030         * @param message
031         *            The error message.
032         */
033        public ValidatorError(ValidationArgs<T> args, String message) {
034                super();
035                _args = args;
036                _message = message;
037        }
038
039        /**
040         * Get the validation arguments.
041         * 
042         * @return The validation arguments.
043         */
044        public ValidationArgs<T> getArgs() {
045                return _args;
046        }
047
048        /**
049         * Set validation arguments.
050         * 
051         * @param args
052         *            The validation arguments.
053         */
054        public void setArgs(ValidationArgs<T> args) {
055                this._args = args;
056        }
057
058        /**
059         * Get the error message.
060         * 
061         * @return The error.
062         */
063        public String getMessage() {
064                return _message;
065        }
066
067        /**
068         * Set the error message.
069         * 
070         * @param message
071         *            The error.
072         */
073        public void setMessage(String message) {
074                this._message = message;
075        }
076
077}
