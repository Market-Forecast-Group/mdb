011package org.mfg.mdb.runtime;
012
013import java.io.IOException;
014
015/**
016 * Common interface for all sequential cursors.
017 * 
018 * @author arian
019 * @param <T>
020 *            The generated record class.
021 * 
022 */
023public interface ISeqCursor<T extends IRecord> extends ICursor<T> {
024
025        /**
026         * Move the cursor to the next position.
027         * 
028         * @return <code>true</code> if there are more records.
029         * @throws IOException
030         *             If there is any problem reading the files.
031         */
032        public boolean next() throws IOException;
033
034        /**
035         * <p>
036         * Reset the cursor to retrieve the data from the <code>start</code> to the
037         * <code>stop</code> position.
038         * </p>
039         * <p>
040         * Use this method if you want to reuse the same cursor resources, actually
041         * we recommend to do that when possible. Internally MDB uses this method to
042         * reuse the cursors or the "select" methods.
043         * </p>
044         * <p>
045         * Very important, you cannot reset a closed cursor. Close a cursor when you
046         * will not use it anymore.
047         * </p>
048         * 
049         * @param start
050         *            The cursor statrs here.
051         * @param stop
052         *            The cursor stops here.
053         * @throws IOException
054         *             If there is any problem accessing the files.
055         */
056        public void reset(long start, long stop) throws IOException;
057}
