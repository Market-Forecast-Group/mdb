011package org.mfg.mdb.runtime;
012
013import java.io.IOException;
014
015/**
016 * Common interface for all random cursors.
017 * 
018 * @author arian
019 * @param <T>
020 *            The generated cursor class.
021 * 
022 */
023public interface IRandomCursor<T extends IRecord> extends ICursor<T> {
024
025        /**
026         * Move the cursor to the given <code>position</code> and fetch the data.
027         * 
028         * @param position
029         *            The position to move.
030         * @throws IOException
031         *             If there is any problem reading the files.
032         */
033        public void seek(long position) throws IOException;
034
035        /**
036         * Move the cursor to the last position and fetch the data.
037         * 
038         * @throws IOException
039         *             If there is any problem reading the files.
040         */
041        public void seekLast() throws IOException;
042
043        /**
044         * Move the cursor to the first position and fetch the data.
045         * 
046         * @throws IOException
047         *             If there is any problem reading the files.
048         */
049        public void seekFirst() throws IOException;
050}
