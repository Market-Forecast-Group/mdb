018/**
019 * <p>
020 * This is the main class of MDB, use this to define a database schema and
021 * compile it.
022 * <p>
023 * You can extend this class and call the schema methods like a script:
024 * </p>
025 * 
026 * <pre>
027 * class Demo {
028 *      Demo() {
029 *              schemaName(&quot;Stock&quot;);
030 *              packageName(&quot;com.stock.mdb&quot;);
031 * 
032 *              table(&quot;Price&quot;);
033 *              column(&quot;time&quot;, Type.LONG, Order.ASCENDING);
034 *              column(&quot;price&quot;, Type.LONG);
035 *      }
036 * 
037 *      public static void main(String[] args) {
038 *              Demo demo = new Demo();
039 *              demo.compile(&quot;/path/to/demo/source&quot;);
040 *      }
041 * }
042 * </pre>
043 * <p>
044 * However, if you like to use a visual tool to define and compile the schema
045 * you may be interested in the {@link SchemaEditor} class.
046 * </p>
047 * 
048 * @see SchemaEditor
049 * @author arian
050 * @author Antonio de Rojas
051 * 
052 */

207        /**
208         * Define a column. In this example we define the a table A with columns a1,
209         * a2, and a table B with columns b1 and b2:
210         * 
211         * <pre>
212         * table("A");
213         *      column("a1", ...);
214         *      column("a2", ...);
215         * 
216         * table("B");
217         *      column("b1");
218         *      column("b2");
219         * </pre>
220         * 
221         * @param name
222         *            The name. Use a valid Java identifier, this one is used as
223         *            part of methods and fields names in the generated code.
224         * @param type
225         *            The type.
226         * @param order
227         *            The order.
228         * @param virtual
229         *            If the column is virtual.TODO:Missing reference.
230         * @param formula
231         *            The formula to compute the value, or <code>null</code> in case
232         *            the column is not virtual.TODO:Missing reference.
233         * @param uuid
234         *            The unique identifier.
235         */
236        public void column(final String name, final Type type, final Order order,
237                        final boolean virtual, final String formula, UUID uuid) {
238                if (_lastTable == null) {
239                        throw new IllegalArgumentException(
240                                        "There not any stable defined yet");
241                }
242                Column col = new Column(name, type, order, virtual, formula, _offset,
243                                uuid);
244                col.setIndex(_lastTable.size());
245                _lastTable.add(col);
246                if (!virtual) {
247                        _offset += type.getSize();
248                }
249        }

386        /**
387         * Generate the MDB classes. Be careful, if the file exists they will be
388         * overwritten, so do not write custom code on MDB files, unless you respect
389         * the "User Code Sections".TODO:Missing reference.
390         * 
391         * @param srcDir
392         *            Use this destination instead of the one the provided in the
393         *            constructor.
394         * 
395         * @param listener
396         *            The compiler listener.
397         * 
398         * @throws IOException
399         *             If there is any error writing the files.
400         */
401        public void compile(String srcDir, final ICompilerListener listener)
402                        throws IOException {
403                if (srcDir == null) {
404                        throw new IllegalArgumentException("Null source directory.");
405                }
406                if (_packageName == null) {
407                        throw new IllegalArgumentException("Null package name.");
408                }
409
410                String srcDir2 = srcDir;
411                if (!srcDir2.endsWith("/")) {
412                        srcDir2 += "/";
413                }
414                // compute the target dir path
415                final String dirPath = srcDir2 + _packageName.replaceAll("\\.", "/");
416
417                for (final Table table : _schema) {
418                        // creates the target file
419                        final File file = new File(dirPath, table.getName() + "MDB.java");
420
421                        final StringBuilder builder = new StringBuilder();
422                        // build the code associated to the table
423                        buildTable(builder, table, file);
424
425                        writeFile(listener, builder, file);
426                }
427
428                buildSession(listener, dirPath);
429        }
430
431        private void buildSession(final ICompilerListener listener,
432                        final String dirPath) throws IOException, FileNotFoundException {
433                File sessionFile = new File(dirPath, _schema.getName()
434                                + "MDBSession.java");
435
436                try (final ByteArrayOutputStream output = new ByteArrayOutputStream()) {
437
438                        final SessionClassGenerator gen = new SessionClassGenerator();
439
440                        try (OpenPrinter printer = new OpenPrinter(output,
441                                        sessionFile)) {
442                                printer.setRegistry(_registry);
443                                gen.emitter = printer;
444                                gen.schema = _schema;
445                                gen.pkgname = _packageName;
446                                gen.run();
447                        }
448
449                        String str = output.toString();
450
451                        StringBuilder builder = new StringBuilder(str);
452
453                        writeFile(listener, builder, sessionFile);
454                }
455        }
456
457        private static void writeFile(final ICompilerListener listener,
458                        final StringBuilder builder, final File file)
459                        throws FileNotFoundException, IOException {
460                // creates the file parents
461                file.getParentFile().mkdirs();
462
463                // print a message
464                if (file.exists()) {
465                        listener.compiledFile(file, true);
466                        // delete in case of file exists
467                        file.delete();
468                } else {
469                        listener.compiledFile(file, false);
470                }
471
472                // write the code to the file
473                try (FileOutputStream stream = new FileOutputStream(file)) {
474                        stream.write(builder.toString().getBytes());
475                }
476        }

478        /**
479         * Build the codes associated to a table. This method fill the templates
480         * 'class.java.template' and 'indexMethods.java.template' and append the
481         * text to the builder.
482         * 
483         * @param builder
484         *            A builder.
485         * @param table
486         *            A table.
487         * @param targetfile
488         * @throws IOException
489         */
490        private void buildTable(final StringBuilder builder, final Table table,
491                        final File targetFile) throws IOException {
492                try (ByteArrayOutputStream output = new ByteArrayOutputStream()) {
493                        try (final OpenPrinter emitter = new OpenPrinter(output, targetFile)) {
494                                emitter.setRegistry(_registry);
495                                MDBClassGenerator gen = new MDBClassGenerator();
496                                gen.emitter = emitter;
497                                gen.table = table;
498                                gen.pkgname = _packageName;
499                                gen.bufferSize = _bufferSize;
500                                gen.run();
501                        }
502                        builder.append(output.toString());
503                }
504        }
