001package org.mfg.mdb.compiler;
002
003import java.io.File;
004
005/**
006 * An interface to listen the files created by the compiler. External tools can
007 * use this to show a progress of the compilation.
008 * 
009 * @author arian
010 * 
011 */
012public interface ICompilerListener {
013        /**
014         * Method called when a file is created.
015         * 
016         * @param file
017         *            The file is created.
018         * @param replacing
019         *            If it is replacing an old file.
020         */
021        public void compiledFile(File file, boolean replacing);
022}
