001package org.mfg.mdb.compiler;
002
003/**
004 * The extension targets are the points of an MDB class where the user can
005 * inject custom code via a compiler extension.
006 * 
007 * @see ICompilerExtension
008 * @author arian
009 */
010public interface IExtensionTargets {
011        /**
012         * Refers to the body of an MDB class. You can use this to add custom MDB
013         * classes members like fields, methods, etc...
014         */
015        public String TARGET_MDB_CLASS = "mdb";
016
017        /**
018         * Refers to the end of the <code>append()</code> methods of an MDB class.
019         * Use this target to add custom statements at the end of the method. See
020         * the example of the {@link ICompilerExtension} javadoc.
021         */
022        public String TARGET_MDB_CLASS_APPEND_METHOD = "mdb.appendMethod";
023
024        /**
025         * Refers to the body of an MDB session class. Use this to add custom
026         * members like methods, fields, etc...
027         */
028        public String TARGET_SESSION_CLASS = "session";
029
030}
