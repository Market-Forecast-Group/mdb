269        private Schema openSchemaFromFile() {
270                final String lastOpenPath = _prefsNode.get("openPath", null);
271                if (lastOpenPath != null) {
272                        _fileChooser.setCurrentDirectory(new File(lastOpenPath));
273                }
274                if (_fileChooser.showOpenDialog(null) == JFileChooser.APPROVE_OPTION) {
275                        try {
276                                final File file = _fileChooser.getSelectedFile();
277                                _prefsNode.put("openPath", file.getAbsolutePath());
278
279                                try (FileReader fileReader = new FileReader(file)) {
280                                        JSONObject jsonOBJ = new JSONObject(new JSONTokener(
281                                                        fileReader));
282                                        final Schema schema = new Schema(jsonOBJ);
283                                        return schema;
284                                }
285                        } catch (final Exception e1) {
286                                JOptionPane.showMessageDialog(null, "Error:" + e1.getMessage());
287                                e1.printStackTrace();
288                        }
289                }
290                return null;
291        }

535        private AbstractTableModel createColumnsModel() {
536                return new AbstractTableModel() {
537
538                        private static final long serialVersionUID = 1L;
539                        private static final int COL_NAME = 0;
540                        private static final int COL_TYPE = 1;
541                        private static final int COL_ORDER = 2;
542                        private static final int COL_VIRTUAL = 3;
543                        private static final int COL_FORMULA = 4;
544
545                        @Override
546                        public int getRowCount() {
547                                final Table selectedTable = getSelectedTable();
548                                return selectedTable == null ? 0 : selectedTable.size();
549                        }
550
551                        @Override
552                        public int getColumnCount() {
553                                return 5;
554                        }
555
556                        @Override
557                        public boolean isCellEditable(final int rowIndex,
558                                        final int columnIndex) {
559                                if (columnIndex == COL_FORMULA) {
560                                        return ((Boolean) getValueAt(rowIndex, COL_VIRTUAL))
561                                                        .booleanValue();
562                                }
563                                return true;
564                        }
565
566                        @Override
567                        public void setValueAt(final Object aValue, final int rowIndex,
568                                        final int columnIndex) {
569                                final Column col = getSelectedTable().get(rowIndex);
570                                switch (columnIndex) {
571                                case COL_NAME:
572                                        col.setName((String) aValue);
573                                        break;
574                                case COL_TYPE:
575                                        col.setType((Type) aValue);
576                                        break;
577                                case COL_ORDER:
578                                        col.setOrder((Order) aValue);
579                                        break;
580                                case COL_VIRTUAL:
581                                        col.setVirtual(((Boolean) aValue).booleanValue());
582                                        break;
583                                case COL_FORMULA:
584                                        col.setFormula((String) aValue);
585                                }
586                        }
587
588                        @Override
589                        public String getColumnName(final int column) {
590                                switch (column) {
591                                case COL_NAME:
592                                        return "Name";
593                                case COL_TYPE:
594                                        return "Type";
595                                case COL_ORDER:
596                                        return "Order";
597                                case COL_VIRTUAL:
598                                        return "Virtual";
599                                case COL_FORMULA:
600                                        return "Formula";
601                                }
602                                return null;
603                        }
604
605                        @Override
606                        public Class<?> getColumnClass(final int columnIndex) {
607                                switch (columnIndex) {
608                                case COL_NAME:
609                                        return String.class;
610                                case COL_TYPE:
611                                        return Type.class;
612                                case COL_ORDER:
613                                        return Order.class;
614                                case COL_VIRTUAL:
615                                        return Boolean.class;
616                                case COL_FORMULA:
617                                        return String.class;
618                                }
619                                return null;
620                        }
621
622                        @Override
623                        public Object getValueAt(final int rowIndex, final int columnIndex) {
624                                final Column col = getSelectedTable().get(rowIndex);
625                                switch (columnIndex) {
626                                case COL_NAME:
627                                        return col.getName();
628                                case COL_TYPE:
629                                        return col.getType();
630                                case COL_ORDER:
631                                        return col.getOrder();
632                                case COL_VIRTUAL:
633                                        return Boolean.valueOf(col.isVirtual());
634                                case COL_FORMULA:
635                                        return col.getFormula();
636                                }
637                                return null;
638                        }
639                };
640        }

651        void save() {
652                if (_lastSavedFile != null) {
653                        _fileChooser.setSelectedFile(_lastSavedFile);
654                }
655                if (_fileChooser.showSaveDialog(null) == JFileChooser.APPROVE_OPTION) {
656                        File file = _fileChooser.getSelectedFile();
657                        try {
658
659                                // String extension = ".mdbs";
660                                final String extension = ".schema-json";
661
662                                if (!file.getName().endsWith(extension)) {
663                                        file = new File(file.getAbsolutePath() + extension);
664                                }
665                                file.createNewFile();
666                                _lastSavedFile = file;
667                                out.println("Save to " + file.getAbsolutePath());
668
669                                updateSchemaFromUI();
670                                try (final FileWriter w = new FileWriter(file)) {
671                                        final String str = _schema.toJSONString();
672                                        w.write(new JSONObject(str).toString(4));
673                                } catch (final JSONException e) {
674                                        e.printStackTrace();
675                                        JOptionPane.showMessageDialog(null,
676                                                        "Error:" + e.getMessage());
677                                }
678                        } catch (final IOException e1) {
679                                JOptionPane.showMessageDialog(null, "Error:" + e1.getMessage());
680                                e1.printStackTrace();
681                        }
682
683                }
684
685                try {
686                        _prefsNode.flush();
687                } catch (final BackingStoreException e) {
688                        e.printStackTrace();
689                }
690        }

747        private Compiler buildCompiler() {
748                final Compiler compiler = new Compiler(_schemaNameTextField.getText(),
749                                new File(_rootDirTextField.getText(), _srcTextField.getText())
750                                                .getAbsolutePath(), _packageNameTextField.getText(),
751                                Integer.parseInt(_bufferSizeTextField.getText()));
752
753                for (final Table t : _schema) {
754                        compiler.table(t.getName(), t.getUUID());
755                        for (final Column col : t) {
756                                compiler.column(col.getName(), col.getType(), col.getOrder(),
757                                                col.isVirtual(), col.getFormula(), col.getUUID());
758                        }
759                }
760                return compiler;
761        }
