001package org.mfg.mdb.compiler;
002
003/**
004 * The order of a column. The MDB runtime assumes all the
005 * values of this columns will accomplish the given order. The order is used to
006 * generate "indexOf" methods that perform binary searches to find the index of
007 * a value. You can see this like the equivalent of the indexes in SQL systems.
008 * 
009 * @author arian
010 * 
011 */
012public enum Order {
013        /**
014         * Ascending order.
015         */
016        ASCENDING,
017
018        /**
019         * Descending order.
020         */
021        DESCENDING,
022
023        /**
024         * No order, no "indexOf" methods are generated for this column.
025         */
026        NONE;
027
028        /**
029         * A shorter method to know if this is {@link #NONE}.
030         * 
031         * @return If this is {@link #NONE}.
032         */
033        public boolean isNone() {
034                return this == NONE;
035        }
036}
